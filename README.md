## requirements 
The requirements are as follow for windows

    pip install sockets

    pip install time

    pip install cython

    pip install --upgrade pip wheel setuptools virtualenv

    virtualenv kivy_env

    kivy_venv\Scripts\activate

    pip install docutils pygments pypiwin32 kivy_deps.sdl2==0.1.* kivy_deps.glew==0.1.*

    pip install kivy

    pip install kivymd

    pip install sqlite3
